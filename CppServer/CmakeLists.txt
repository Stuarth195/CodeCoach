cmake_minimum_required(VERSION 3.10)
project(CodeCoachServer)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Encontrar cURL
find_package(PkgConfig REQUIRED)
pkg_check_modules(CURL REQUIRED libcurl)

# Archivos fuente
set(SOURCES
    main.cpp
    RequestHandler.cpp
)

# Ejecutable principal
add_executable(CodeCoachServer ${SOURCES})

# Enlazar con cURL
target_link_libraries(CodeCoachServer ${CURL_LIBRARIES})
target_include_directories(CodeCoachServer PRIVATE ${CURL_INCLUDE_DIRS})
