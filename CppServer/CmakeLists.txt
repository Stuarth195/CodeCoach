cmake_minimum_required(VERSION 3.10)
project(CodeCoachServer)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Configuraciones específicas por plataforma
if(WIN32)
    set(PLATFORM_LIBS ws2_32)
else()
    set(PLATFORM_LIBS pthread)
endif()



# Archivos fuente
set(SOURCES
    main.cpp
    RequestHandler.cpp
    RequestHandler.h
)

# Ejecutable principal
add_executable(CodeCoachServer ${SOURCES})
target_link_libraries(CodeCoachServer ${PLATFORM_LIBS})

# Incluir header-only libs (nlohmann::json)
# Se espera que hayas puesto el single-header en ../include/nlohmann/json.hpp
target_include_directories(CodeCoachServer PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/include)

# Añadir subdirectorio runner (compilará runner/runner.exe)
add_subdirectory(runner)
